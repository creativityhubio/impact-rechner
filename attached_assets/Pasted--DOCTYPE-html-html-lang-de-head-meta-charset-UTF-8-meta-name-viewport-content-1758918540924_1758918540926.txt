        // Utility functions
        function formatNumber(num) {
            return Math.round(num).toLocaleString('de-DE');
        }

        function animateValue(element, start, end, duration) {
            const startTime = performance.now();
            const change = end - start;

            function updateValue(currentTime) {
                const elapsed = currentTime - startTime;
                const progress = Math.min(elapsed / duration, 1);
                const easeOutQuart = 1 - Math.pow(1 - progress, 4);
                const current = start + (change * easeOutQuart);

                element.textContent = formatNumber(current);

                if (progress < 1) {
                    requestAnimationFrame(updateValue);
                }
            }

            requestAnimationFrame(updateValue);
        }

        // Main calculation function
        function updateCalculations(animate = false) {
            const investment = parseFloat(document.getElementById('investment-slider').value);

            // Update amount display
            const amountDisplay = document.getElementById('amountDisplay');
            if (animate) {
                const currentAmount = parseFloat(amountDisplay.textContent.replace(/\./g, '')) || 100000;
                animateValue(amountDisplay, currentAmount, investment, 800);
            } else {
                amountDisplay.textContent = formatNumber(investment);
            }

            // Calculate impacts
            const protectedArea = investment * 0.4;
            const protectedFootball = (protectedArea / 7140).toFixed(1);
            const co2Bound = Math.round(protectedArea / 228);
            const co2Equivalent = Math.round(co2Bound / 10);
            const bioProduction = Math.round(investment / 4);
            const jobs = Math.ceil(investment / 20000);
            const animals = Math.round(protectedArea / 80);
            const localValue = Math.round(investment * 0.2);
            const trees5y = Math.round(investment / 80);
            const totalCo210y = co2Bound * 10;

            // Update impact values with smooth animation
            const updates = [
                { id: 'protected-area', value: protectedArea },
                { id: 'protected-football', value: protectedFootball, isDecimal: true },
                { id: 'co2-bound', value: co2Bound },
                { id: 'co2-equivalent', value: co2Equivalent },
                { id: 'bio-production', value: bioProduction },
                { id: 'jobs', value: jobs },
                { id: 'animals', value: animals },
                { id: 'local-value', value: localValue },
                { id: 'trees-5y', value: trees5y },
                { id: 'total-co2-10y', value: totalCo210y }
            ];

            updates.forEach(({ id, value, isDecimal }) => {
                const element = document.getElementById(id);
                if (element) {
                    if (isDecimal) {
                        element.textContent = value.toString().replace('.', ',');
                    } else if (animate) {
                        const currentValue = parseFloat(element.textContent.replace(/\./g, '')) || 0;
                        animateValue(element, currentValue, value, 1200);
                    } else {
                        element.textContent = formatNumber(value);
                    }
                }
            });

            // KORRIGIERTE BERECHNUNG: Update nur das mittlere Szenario (realistic)
            const pachtRate = 0.08; // 8% Pachtzins
            
            // Berechne die Pachteinnahmen basierend auf dem aktuellen Investitionsbetrag
            const monthlyPacht = Math.ceil((investment * pachtRate) / 12);
            const yearlyPacht = Math.ceil(investment * pachtRate);

            // Update current investment display
            const currentInvestmentElement = document.getElementById('current-investment');
            if (currentInvestmentElement) {
                currentInvestmentElement.textContent = formatNumber(investment);
            }

            // Update nur das mittlere Szenario (deine Investition)
            const realisticMonthlyElement = document.getElementById('realistic-monthly');
            const realisticYearlyElement = document.getElementById('realistic-yearly');

            if (realisticMonthlyElement && realisticYearlyElement) {
                if (animate) {
                    const currentMonthly = parseFloat(realisticMonthlyElement.textContent.replace(/\./g, '')) || 0;
                    const currentYearly = parseFloat(realisticYearlyElement.textContent.replace(/\./g, '')) || 0;
                    animateValue(realisticMonthlyElement, currentMonthly, monthlyPacht, 1200);
                    animateValue(realisticYearlyElement, currentYearly, yearlyPacht, 1200);
                } else {
                    realisticMonthlyElement.textContent = formatNumber(monthlyPacht);
                    realisticYearlyElement.textContent = formatNumber(yearlyPacht);
                }
            }

            // Die anderen beiden Szenarien (100.000‚Ç¨ und 1.000.000‚Ç¨) bleiben unver√§ndert

        }
